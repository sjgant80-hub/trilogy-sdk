§ III · the code

How you'd use it in production.

// 1 · install
npm install @sjgant80/trilogy-sdk

// 2 · import
import { Trilogy } from '@sjgant80/trilogy-sdk';

// 3 · BYOK — bring your own provider key
const ai = new Trilogy({ provider: 'anthropic', apiKey: process.env.CLAUDE_KEY });

// DocMind — extract structured data from any document
const receipt = await ai.docmind.extract(receiptText, { schema: 'receipt' });
// → { vendor, amount, date, category, vat, items: [...] }

// Deep — get a structured research report
const findings = await ai.deep.research(question, { format: 'report', sources: true });
// → { summary, sections: [...], sources: [...] }

// Flux — generate brand-aligned content
const post = await ai.flux.generate({ brief, format: 'linkedin', voice: 'direct' });
// → { content, hook, hashtags }
